<body>

<h2>Student Management System</h2>

<input id="roll" placeholder="Roll Number"><br>
<input id="name" placeholder="Name"><br>
<input id="age" placeholder="Age"><br>
<input id="grade" placeholder="Grade"><br>

<button onclick="addStudent()">Add</button>
<button onclick="searchStudent()">Search</button>
<button onclick="updateStudent()">Update</button>
<button onclick="showStudents()">Show All</button>

<div id="output"></div>

<script>
let students = {
    "101": {name: "Rahul", age: "20", grade: "A"},
    "102": {name: "Priya", age: "19", grade: "B"},
    "103": {name: "Aman", age: "21", grade: "A+"},
    "104": {name: "Sneha", age: "20", grade: "B+"},
    "105": {name: "Kiran", age: "22", grade: "A"}
};

function addStudent() {
    let r = document.getElementById("roll").value;
    if (students[r]) {
        alert("Exists");
        return;
    }
    students[r] = {
        name: document.getElementById("name").value,
        age: document.getElementById("age").value,
        grade: document.getElementById("grade").value
    };
    alert("Added!");
}

function searchStudent() {
    let r = document.getElementById("roll").value;
    let output = document.getElementById("output");
    output.innerHTML = students[r] 
        ? JSON.stringify(students[r]) 
        : "Not found";
}

function updateStudent() {
    let r = document.getElementById("roll").value;
    if (students[r]) {
        let name = document.getElementById("name").value;
        let age = document.getElementById("age").value;
        let grade = document.getElementById("grade").value;

        if (name) students[r].name = name;
        if (age) students[r].age = age;
        if (grade) students[r].grade = grade;

        alert("Updated!");
    } else {
        alert("Not found");
    }
}

function showStudents() {
    let output = document.getElementById("output");
    output.innerHTML = "";
    for (let r in students) {
        output.innerHTML += `<p>${r}: ${JSON.stringify(students[r])}</p>`;
    }
}
</script>
